1.3 Add Custom Install Prompt to index.html

Paste this near the bottom of index.html before </body>:

/* ── PWA Install Prompt ── */
let _deferredPrompt = null;

window.addEventListener('beforeinstallprompt', (e) => {
  e.preventDefault();
  _deferredPrompt = e;
  // Show your custom install banner/button here
  document.getElementById('install-banner')?.classList.remove('hidden');
});

function triggerInstallPrompt() {
  if (!_deferredPrompt) return;
  _deferredPrompt.prompt();
  _deferredPrompt.userChoice.then((choice) => {
    if (choice.outcome === 'accepted') {
      console.log('PWA installed');
      document.getElementById('install-banner')?.classList.add('hidden');
    }
    _deferredPrompt = null;
  });
}

window.addEventListener('appinstalled', () => {
  console.log('App installed successfully');
  document.getElementById('install-banner')?.classList.add('hidden');
});

/* ── Service Worker: register + update detection ── */
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('/sw.js', { scope: '/' })
    .then((reg) => {
      console.log('[SW] Registered:', reg.scope);
      reg.addEventListener('updatefound', () => {
        const newSW = reg.installing;
        newSW.addEventListener('statechange', () => {
          if (newSW.state === 'installed' && navigator.serviceWorker.controller) {
            // New version available — show update toast
            if (typeof toast === 'function') {
              toast('🔄 Update available — restart to apply', 'info', 8000);
            }
          }
        });
      });
    })
    .catch((err) => console.error('[SW] Registration failed:', err));
}

5 — Bubblewrap TWA: Complete Android Build Guide

Prerequisites

  • Node.js 18+ (nodejs.org)
  • Java JDK 17+ (adoptium.net)
  • Your Firebase-hosted URL live (e.g. https://stratix-lite-17.web.app)
  • SHA-256 cert fingerprint (generated in Step 4 below)

Step 1 — Install Bubblewrap CLI

npm install -g @bubblewrap/cli
# Verify installation
bubblewrap --version

Step 2 — Create a project folder and initialise TWA

mkdir stratix-twa && cd stratix-twa
bubblewrap init --manifest https://stratix-lite-17.web.app/manifest.json

Answer the prompts like this:

Domain:               stratix-lite-17.web.app
Application name:     Stratix - Business Manager
Short name:           Stratix
Package ID:           com.stratixlite.app
App version code:     1
App version name:     1.0.0
Display:              standalone
Orientation:          portrait
Theme colour:         #F7C948
Background colour:    #070810

Icon URL:             https://stratix-lite-17.web.app/icons/icon-512.png
Maskable icon URL:    https://stratix-lite-17.web.app/icons/icon-maskable-512.png
Signing key path:     press Enter for default (~/.android/stratix.keystore)
Signing key alias:    stratix-key

Step 3 — Generate a signing key (first time only)

# Bubblewrap auto-generates one if you accept defaults.
# Or generate manually with keytool:
keytool -genkeypair \
  -v \
  -keystore stratix-release.keystore \
  -alias stratix-key \
  -keyalg RSA \
  -keysize 2048 \
  -validity 10000 \
  -storepass YOUR_STORE_PASSWORD \
  -keypass YOUR_KEY_PASSWORD \
  -dname "CN=Stratix, OU=App, O=StratixLite, L=Mumbai, ST=Maharashtra, C=IN"

# ⚠️  BACK UP YOUR KEYSTORE FILE IMMEDIATELY
# If you lose it, you can NEVER update the app on Play Store
# Store it in: Google Drive + external hard drive + password manager

Step 4 — Get SHA-256 fingerprint

bubblewrap fingerprint list
# Output looks like:

# Alternative with keytool directly:
keytool -list -v \
  -keystore stratix-release.keystore \
  -alias stratix-key \
  -storepass YOUR_STORE_PASSWORD

6.2 Firestore Security — Status: ✅ Correct

The existing rules are correct. No changes needed:

// firestore.rules — VERIFIED CORRECT
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId}/{document=**} {
      allow read, write: if request.auth != null
                         && request.auth.uid == userId;
    }
    match /{document=**} {
      allow read, write: if false;  // block all other paths
    }
  }
}
Consider adding field-level validation: request.resource.data.amount is number && request.resource.data.amount > 0 to prevent data corruption.

6.3 Razorpay Payment Security

Current state: Payments are verified only on the client side via Razorpay's handler callback. A determined user could manipulate this.

For production, add server-side verification using a Firebase Function:

// functions/index.js
const functions = require('firebase-functions');
const Razorpay  = require('razorpay');
const crypto    = require('crypto');

exports.verifyPayment = functions.https.onCall(async (data, context) => {
  if (!context.auth) throw new functions.https.HttpsError('unauthenticated', 'Login required');

  const { razorpay_order_id, razorpay_payment_id, razorpay_signature } = data;
  const secret  = functions.config().razorpay.secret;
  const hmac    = crypto.createHmac('sha256', secret);
  hmac.update(razorpay_order_id + '|' + razorpay_payment_id);
  const expected = hmac.digest('hex');

  if (expected !== razorpay_signature) {
    throw new functions.https.HttpsError('permission-denied', 'Payment verification failed');
  }

  // Mark user as Pro in Firestore
  await admin.firestore()
    .doc(`users/${context.auth.uid}/kv/plan`)
    .set({ isPro: true, paymentId: razorpay_payment_id, ts: Date.now() });

  return { success: true };
});

# Set your Razorpay secret:
firebase functions:config:set razorpay.secret="YOUR_RAZORPAY_SECRET_KEY"
firebase deploy --only functions

2. Click your Firebase Web API key
3. Under "Application restrictions" → select "HTTP referrers"
4. Add:  stratix-lite-17.web.app/*
         *.stratix-lite-17.firebaseapp.com/*
5. Under "API restrictions" → Restrict to Firebase APIs only
6. Save
